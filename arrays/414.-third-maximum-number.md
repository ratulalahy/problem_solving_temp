```python
class Solution:
    def thirdMax(self, nums: List[int]) -> int:
        max_ary = [-float(inf)]*3
        for cur in nums:
            if not cur in max_ary:
                if cur > max_ary[0]:
                    max_ary = [cur, max_ary[0] ,max_ary[1]]
                elif cur > max_ary[1]:
                    max_ary = [max_ary[0], cur, max_ary[1]]
                elif cur > max_ary[2]:
                    max_ary[2] = cur
        return max_ary[2] if max_ary[2] != -float(inf) else max(max_ary[:2])
```
