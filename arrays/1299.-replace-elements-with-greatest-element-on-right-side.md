# ðŸŸ¢ 1299. Replace Elements with Greatest Element on Right Side

```python
class Solution:
    def replaceElements(self, arr: List[int]) -> List[int]:
        cur_max = arr[-1]
        arr[-1] = -1
        cur_indx = len(arr) - 2
        while(cur_indx >= 0):
            if arr[cur_indx] > cur_max:
                tmp = arr[cur_indx]
                arr[cur_indx] = cur_max
                cur_max = tmp
            else:
                arr[cur_indx] = cur_max
            cur_indx -= 1
        return arr

```
