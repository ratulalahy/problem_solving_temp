class Solution:
    def removeElement(self, nums: List[int], val: int) -> int:
        ptr_val = 0
        ptr_all = 0
        
        while(ptr_all < len(nums)):
            if nums[ptr_all] != val:
                nums[ptr_val] = nums[ptr_all]
                ptr_all += 1
                ptr_val += 1
            else:
                ptr_all +=1
        return ptr_val
