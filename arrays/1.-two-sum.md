```python
class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        hash_map = {}
        for indx, val in enumerate(nums):
            cur_tar = target-val
            if target-val in hash_map:
                return [hash_map[cur_tar], indx]
            else:
                hash_map[val] = indx
        raise Exception("Not found!")
```
