```python
class Solution:
    def validMountainArray(self, arr: List[int]) -> bool:
        if len(arr) < 3:
            return False
        
        cur_indx = 1
        while((cur_indx < len(arr)) and (arr[cur_indx-1] < arr[cur_indx])):
            cur_indx += 1
        if (cur_indx == 1) or (cur_indx == len(arr)):
            print(cur_indx)
            return False
        while((cur_indx < len(arr)) and (arr[cur_indx-1] > arr[cur_indx])):
            cur_indx += 1
        
        return cur_indx == len(arr)
```
