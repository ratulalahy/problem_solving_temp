# ðŸŸ  707. Design Linked List

```python
class Node:
    def __init__(self, val):
        self.val = val
        self.next = None

class MyLinkedList:

    def __init__(self):
        self.head = None
        self.size = 0

    def get(self, index: int) -> int:
        if index > self.size-1:
            return -1
        if self.size == 0:
            return -1
        
        cur_indx = 0
        cur_node = self.head
        while (cur_indx < index):
            cur_node = cur_node.next
            cur_indx += 1
        return cur_node.val

    def addAtHead(self, val: int) -> None:
        cur_node = Node(val)
        cur_node.next = self.head
        self.head = cur_node
        self.size += 1

    def addAtTail(self, val: int) -> None:
        cur_node = self.head
        if cur_node == None:
            self.head = Node(val)
            self.size += 1
            return
        while(cur_node.next != None):
            cur_node = cur_node.next
        cur_node.next = Node(val)
        self.size += 1

    def addAtIndex(self, index: int, val: int) -> None:
        if index > self.size:
            return -1
        if index == 0:
            self.addAtHead(val)
        else:
            cur_node = self.head
            cur_indx = 0
            while(cur_indx < index-1):
                cur_node = cur_node.next
                cur_indx += 1
            new_node = Node(val)
            new_node.next = cur_node.next
            cur_node.next = new_node
        self.size += 1

    def deleteAtIndex(self, index: int) -> None:
        if index >= self.size:
            return None
        elif index == 0:
            self.head = self.head.next
        else:
            cur_node = self.head
            cur_indx = 0
            while(cur_indx < index-1):
                cur_node = cur_node.next
                cur_indx += 1
            cur_node.next = cur_node.next.next
        self.size -= 1


# Your MyLinkedList object will be instantiated and called as such:

# obj.addAtHead(val)
# obj.addAtTail(val)
# obj.addAtIndex(index,val)
# obj.deleteAtIndex(index)
```
